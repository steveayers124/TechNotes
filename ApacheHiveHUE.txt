LEFT SEMI JOIN is like ANSI SQL clause EXISTS
--------------
# ANSI SQL
SELECT *
FROM a
WHERE EXISTS (
  SELECT *
  FROM b
  WHERE a.name = b.name
)
# HIVE
SELECT *
FROM a
LEFT SEMI JOIN b ON a.name = b.name
